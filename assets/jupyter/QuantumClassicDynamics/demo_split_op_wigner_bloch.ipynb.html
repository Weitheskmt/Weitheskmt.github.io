# parameters of the quantum system
qsys_params = dict(
    t=0.,
    dt=0.05,

    x_grid_dim=256,
    x_amplitude=10.,

    p_grid_dim=256,
    p_amplitude=10.,

    beta=1. / 0.7,

    # kinetic energy part of the hamiltonian
    k=lambda p: 0.5 * p ** 2,

    # potential energy part of the hamiltonian
    v=lambda x: 0.5 * x ** 4,
)
